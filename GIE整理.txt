-- GIT 下载
	git init
		-- 初始化本地仓库

	git clone 地址
		-- 克隆项目

-- GIT 更新

	git status
		-- 命令用于显示工作目录和暂存区的状态。使用此命令能看到那些修改被暂存到了, 哪些没有, 哪些文件没有被Git tracked到

	git fetch origin master:tmp
		-- //从远程仓库master分支获取最新，在本地建立tmp分支

	git diff tmp
		-- 前分支和tmpM行Ρ
	
	git merge tmp
		-- 合并tmp分支到当前分支

-- GIT 提交

	git add -u 路径
		-- 将状态改变的代码提交至缓存

	git add -A 路径
		-- 将所有的已跟踪的文件的修改与删除和新增的未跟踪的文件都添加到暂存区。

	git commit -m ""
		-- 将代码提交到本地仓库中 

	git push
		-- 将缓存区代码推送到Git服务器

-- GIT 删除
	
	rm -rf .git
		-- 删除本地仓库

-- GIT 查看
	
	git branch
		-- 显示所有本地分支

	git branch -a
		-- 显示所有线上分支

-- GIT 切换分支
	git checkout -b 名字 origin/名字
		-- 切换分支

-- GIT 回退版本
	git reset --hard 版本号

-- GIT 强制提交
	git push -f -u origin 分支
